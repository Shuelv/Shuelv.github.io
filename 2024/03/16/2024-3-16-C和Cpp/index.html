int main()
{
int a = 20; int b = 4; int c = 2; int d = 1;
printf(“a++=%d\n”, a++);
printf(“a=%d\n”, a);
printf(“++a=%d\n”, ++a);
printf(“a=%d\n”, a);
printf(“a—=%d\n”, a—);
printf(“a=%d\n”, a);
printf(“—a=%d\n”, —a);
printf(“a=%d\n”, a);
/a后面++或—，这个表达式(如a++)的值不变，运行后的值(a)改变;
a前面++或—，这个表达式的值和运行后的值均改变/
printf(“b-a/(c+2)=%d\n”, b - a / (c+2));
//算术运算符中乘除优先于加减，括号最优先
printf(“a/=b-c=%d\n”,a/=b-c);
//复合赋值运算时先算=号右边
printf(“a-b==c>d=%d\n”, a - b == c > d);
//关系运算符中不等性比较(>,<,>=,<=)大于相等性比较(==,!=)
//算术>移位>关系>逻辑
//碰到赋值(=)就先算右边
return 0;
}
